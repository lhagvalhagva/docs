## Endpoint

<RequestExample>
```bash cURL
curl -X GET 'https://your-odoo-instance.com/api/hr-leave/time-requests' \
  -H 'Cookie: session_id=your_session_id'
```
</RequestExample>

## Response Fields

<ResponseField name="success" type="boolean" required>
Амжилттай эсэх
</ResponseField>

<ResponseField name="data" type="array">

<Expandable title="Time Request">
  <ResponseField name="id" type="integer" required>

  </ResponseField>

  <ResponseField name="employeeId" type="integer">
  Ажилтны ID
  </ResponseField>

  <ResponseField name="leaveTypeId" type="integer">

  </ResponseField>

  <ResponseField name="leaveTypeName" type="string">

  </ResponseField>

  <ResponseField name="leaveFlowName" type="string">

  <ResponseField name="dateFrom" type="string">
